package cp2022.solution;

public class Pair< T, W > {
    private final T first;
    private final W second;

    public Pair(T t, W w) {
        this.first = t;
        this.second = w;
    }

    public T getFirst() {
        return first;
    }

    public W getSecond() {
        return second;
    }
}
